00066 void QQuickAbstractDialog::setVisible(bool v)
00067 {
00068     if (m_visible == v) return;
00069     m_visible = v;
00070     if (helper()) {
00071         if (v) {
00072             Qt::WindowFlags flags = Qt::Dialog;
00073             if (!title().isEmpty())
00074                 flags |= Qt::WindowTitleHint;
00075             m_visible = helper()->show(flags, m_modality, parentWindow());
00076         } else {
00077             helper()->hide();
00078         }
00079     } else {
00080         // For a pure QML implementation, there is no helper.
00081         // But m_implementation is probably either an Item or a Window at this point.
00082         if (!m_dialogWindow) {
00083             m_dialogWindow = qobject_cast<QWindow *>(m_qmlImplementation);
00084             if (!m_dialogWindow) {
00085                 QQuickItem *dlgItem = qobject_cast<QQuickItem *>(m_qmlImplementation);
00086                 if (dlgItem) {
00087                     m_dialogWindow = dlgItem->window();
00088                     // An Item-based dialog implementation doesn't come with a window, so
00089                     // we have to instantiate one iff the platform allows it.
00090                     if (!m_dialogWindow && QGuiApplicationPrivate::platformIntegration()->
00091                             hasCapability(QPlatformIntegration::MultipleWindows)) {
00092                         QQuickWindow *win = new QQuickWindow;
00093                         ((QObject *)win)->setParent(this); // memory management only
00094                         m_dialogWindow = win;
00095                         dlgItem->setParentItem(win->contentItem());
00096                         m_dialogWindow->setMinimumSize(QSize(dlgItem->width(), dlgItem->height()));
00097                         connect(win, SIGNAL(widthChanged(int)), this, SLOT(windowGeometryChanged()));
00098                         connect(win, SIGNAL(heightChanged(int)), this, SLOT(windowGeometryChanged()));
00099                     }
00100 
00101                     QQuickItem *parentItem = qobject_cast<QQuickItem *>(parent());
00102                     // qDebug() << "item implementation" << dlgItem << "has window" << m_dialogWindow << "and parent" << parentItem;
00103 
00104                     // If the platform does not support multiple windows, but the dialog is
00105                     // implemented as an Item, then just reparent it and make it visible.
00106                     // TODO QTBUG-29818: put it into a fake Item-based window, when we have a reusable self-decorated one.
00107                     if (parentItem && !m_dialogWindow)
00108                         dlgItem->setParentItem(parentItem);
00109                 }
00110             }
00111             if (m_dialogWindow)
00112                 connect(m_dialogWindow, SIGNAL(visibleChanged(bool)), this, SLOT(visibleChanged(bool)));
00113         }
00114         if (m_dialogWindow) {
00115             if (v) {
00116                 m_dialogWindow->setTransientParent(parentWindow());
00117                 m_dialogWindow->setTitle(title());
00118                 m_dialogWindow->setModality(m_modality);
00119             }
00120             m_dialogWindow->setVisible(v);
00121         }
00122     }
00123 
00124     emit visibilityChanged();
00125 }
