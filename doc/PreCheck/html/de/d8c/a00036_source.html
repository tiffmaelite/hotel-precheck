00104                         h = fract(h);

00109                         else if (h < 2.0 / 3.0)

00115                     highp vec3 HSLtoRGB(highp vec3 color) {
00116                         highp float h = color.x;
00117                         highp float l = color.z;
00118                         highp float s = color.y;
00119 
00120                         if (s < 1.0 / 256.0)
00121                             return vec3(l, l, l);
00122 
00123                         highp float v1;
00124                         highp float v2;
00125                         if (l < 0.5)
00126                             v2 = l * (1.0 + s);
00127                         else
