00168 void QQuickAbstractFileDialog::updateModes()
00169 {
00170     // The 4 possible modes are AnyFile, ExistingFile, Directory, ExistingFiles
00171     // Assume AnyFile until we find a reason to the contrary
00172     QFileDialogOptions::FileMode mode = QFileDialogOptions::AnyFile;
00173 
00174     if (m_selectFolder) {
00175         mode = QFileDialogOptions::Directory;
00176         m_options->setOption(QFileDialogOptions::ShowDirsOnly);
00177         m_selectMultiple = false;
00178         m_selectExisting = true;
00179         setNameFilters(QStringList());
00180     } else if (m_selectExisting) {
00181         mode = m_selectMultiple ?
00182             QFileDialogOptions::ExistingFiles : QFileDialogOptions::ExistingFile;
00183         m_options->setOption(QFileDialogOptions::ShowDirsOnly, false);
00184     } else if (m_selectMultiple) {
00185         m_selectExisting = true;
00186     }
00187     if (!m_selectExisting)
00188         m_selectMultiple = false;
00189     m_options->setFileMode(mode);
00190     m_options->setAcceptMode(m_selectExisting ?
00191                            QFileDialogOptions::AcceptOpen : QFileDialogOptions::AcceptSave);
00192     emit fileModeChanged();
00193 }
